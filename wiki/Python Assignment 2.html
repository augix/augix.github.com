Exercise 5: reverse complementary

<python> #!/usr/bin/env python import sys counts = {} comp = dict(zip('ATGC','TACG')) def revcomp(seq):

seq = list(seq) return ''.join([comp.get(nt,'N') for nt in seq[::-1]])

for line in sys.stdin:
