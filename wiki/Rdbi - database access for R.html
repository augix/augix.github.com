Install

R CMD INSTALL ~/Library/R/ rdbi.tar.gz or <R> > install.packages("rdbi",lib="~/Library/R/")

> library(Rdbi) > library(RdbiPgSQL) > conn <- dbConnect(PgSQL(), host = "10.10.118.118", dbname = "expression") > res <- dbSendQuery(conn,"select * from exonarray.probe limit 10;") > mydata <- dbGetResult(res) > mydata
